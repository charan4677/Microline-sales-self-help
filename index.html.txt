function calcRisk(){
  var score=0;
  var gap=sTGST()-cTGST();
  if(gap>500000)score+=25;else if(gap>250000)score+=15;else if(gap>100000)score+=8;
  score+=D.c.demod*10;
  score+=D.c.ceb*10;
  if(cAnn()<sAnn())score+=15;
  if(D.c.excl>0)score-=5; // Senses advantage
  return Math.max(0, Math.min(score, 100));
}

function renderDash(){

  var wpColor = winProb>=65?'var(--green)':winProb>=40?'var(--amber)':'var(--red)';
  var wpLabel = winProb>=65?'High Probability':winProb>=40?'Evaluating':'At Risk';

  // Render Win Prob Ring
  var arc=document.getElementById('dash-win-arc');
  var num=document.getElementById('dash-win-num');
  var lbl=document.getElementById('dash-win-label');
  if(arc){
    var offset=301.59 - (winProb/100)*301.59;
    arc.setAttribute('stroke-dashoffset',offset);
    arc.setAttribute('stroke',wpColor);
  }
  if(num){num.textContent=winProb+'%'; num.setAttribute('fill',wpColor);}
  if(lbl){lbl.textContent=wpLabel;}

  // Render Next Best Action (NBA)
  var nbaText = document.getElementById('dash-nba-text');
